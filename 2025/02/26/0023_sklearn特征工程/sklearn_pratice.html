import seaborn as sns

# %matplotlib inline 一个notebook中只需要运行一次，
# 之后用matplotlib库作图不需要plt.show()即可把图展示出来

import matplotlib.pyplot as plt
%matplotlib inline
import warnings
warnings.filterwarnings("ignore")

from sklearn.datasets import load_iris
from sklearn.preprocessing import StandardScaler

# load_iris 使用鸢尾花数据集iris
# 特征数据X
X = load_iris().data
# 目标变量
Y = load_iris().target

# 标准化：将数据缩放到均值为0，标准差为1的高斯分布过程。
# 使用 preproccessing 库的StandardScaler类对数据进行标准化
data_stander = StandardScaler().fit_transform(X)
sns.distplot(data_stander,color='red')
data_stander[:5]
